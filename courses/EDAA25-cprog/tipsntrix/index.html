To align stuff:

/*  Called with malloc_aligned(128, 32, &q); */
void *malloc_aligned(size_t n, size_t a, void **ptr)
{
        uintptr_t b;
        void*     p;

        n += a-1;

        *ptr = p = malloc(n);
        b = (uintptr_t)p;
        b += a-1;
        b &= ~(a-1);
        p = (void*) b;

        return p;
}
